\section{Язык высказываний}

Давайте перед началом вспомним, что такое формальный язык и контекстно-свободная грамматика.

Символы --- это значки, которые мы не определяем. Например, $a$, $b$, $c$ --- это символы.
Слова --- это последовательность символов ($abc$ --- это слово). Если мы пишем слова последовательно, то мы конкатенируем эти слова и получаем новое слово (Если $w = ab, v = cd $, то $ wv = abcd$). Мы бронием значок $\epsilon$ для обозначения пустого слова.

Алфавит --- это множество букв. Язык --- это множество слов.

Если $L$ --- язык, то $L^* = \bigcup_{i=0}^\infty L^i$

\begin{definition}
  Контекстно-свободная грамматика $G$ --- это $(V,T,P,S)$, где
  \begin{itemize}
  \item $V$ - мн-во нетерминальных символов,
  \item $T$ - мн-во терминальных символов,
  \item $P$ - мн-во продукций, конкретно, элементов $A \longrightarrow \alpha$, где $A \in V, \alpha \in (V \cup T)^*$
  \item $S \in V$ -  начальный символ

  \end{itemize}
  (Про то, как он порождает язык, я не буду говорить)))) )
\end{definition}

\begin{definition}
  Пусть $\mathcal{P}$ - произвольный алфавит, который мы будем называть множество пропозиональных переменных.

  Тогда $L$ - язык высказываний, если это язык в алфавите $\mathcal{P} \sqcup \{\land, \lor, \lnot, \rightarrow, ), (, \bot, \top\}$ и задаётся следующей грамматикой:
  \begin{align*}
    S & \longrightarrow p \text{, где } p \in \mathcal{P}\\
    & \longrightarrow \bot\\
    & \longrightarrow \top\\
    & \longrightarrow (\lnot S)\\
    & \longrightarrow (S \lor S)\\
    & \longrightarrow (S \land S)\\
    & \longrightarrow (S \to S)\\
  \end{align*}
\end{definition}

Договоримся, что у нас такой приоритет операций
$\lnot\land\lor\to$
, начиная, конечно, с первой операции.

Договоримся, что $\lor$ и $\land$ - левоассоциативные операция.
Насчёт $\to$: математики любят, чтобы они было левоассоциативными
, а программисты --- правоассоциативные. (Та же ассоциативность, что стрелочном типе в система типов HM).



\section{Классическая семантика}
%% \epigraph{\itshape Begin at the beginning, the King said gravely, ``and go on till you come to the end: then stop.''}{---Lewis Carroll, \textit{Alice in Wonderland}}
Приготовьтесь, здесь будет много определений.
  Здесь $\mathbb{B}_0 = \{0, 1\}$

\begin{definition}
  $\mathcal{I}: \mathcal{P} \rightarrow \mathbb{B}_0$ --- интерпретация (Напомним, что $\mathcal{P}$ --- это множество пропозициональных переменных)
\end{definition}

\begin{definition}
  $[\![ \varphi ]\!]_{\mathcal{I}}$ - семантическая скобка, которая выдает значение из $\mathbb{B}_0$, где $\varphi$ - формула
  Индуктивно определим:
  \begin{itemize}
  \item $[\![ p ]\!]_{\mathcal{I}} = \mathcal{I}(p)$, $p \in \mathcal{P}$
  \item $[\![ \bot ]\!]_{\mathcal{I}} = 0$
  \item $[\![ \top ]\!]_{\mathcal{I}} = 1$
  \item $[\![ \lnot \varphi ]\!]_{\mathcal{I}} = 1 - [\![ \varphi ]\!]_{\mathcal{I}}$
  \item $[\![ \varphi \lor \psi ]\!]_{\mathcal{I}} = max([\![ \varphi ]\!]_{\mathcal{I}}, [\![ \psi ]\!]_{\mathcal{I}})$
  \item $[\![ \varphi \land \psi ]\!]_{\mathcal{I}} = min([\![ \varphi ]\!]_{\mathcal{I}}, [\![ \psi ]\!]_{\mathcal{I}})$
  \item $[\![ \varphi \to \psi ]\!]_{\mathcal{I}} = [\![ \lnot \varphi \lor \psi ]\!]_{\mathcal{I}}$
  \end{itemize}
\end{definition}

\begin{example}
  Пусть $\mathcal{I}$ --- такая интерпретация, что $\mathcal{I}(p) = 1, \mathcal{I}(q) = 0$

  Тогда $[\![ (p \to q) \lor (q \to (q \land \lnot q)) ]\!]_{\mathcal{I}} = 1$
\end{example}

\begin{definition}
  \leavevmode
  \begin{enumerate}
  \item $\varphi$ - тавтология, если $\forall \ \mathcal{I}$ --- интерпретация, $[\![ \varphi ]\!]_{\mathcal{I}} = 1$
  \item $\varphi$ - невыполнима (UNSAT), если $\forall \ \mathcal{I}$ --- интерпретация, $[\![ \varphi ]\!]_{\mathcal{I}} = 0$
  \item $\varphi$ - выполнима (SAT), если $\exists \ \mathcal{I}$ --- интерпретация, $[\![ \varphi ]\!]_{\mathcal{I}} = 1$

  \end{enumerate}
\end{definition}
\begin{example}
  В классической логике (которую мы рассматриваем) формула $a \lor \lnot a$ ---  тавтология
\end{example}

\begin{definition}[Логическое следствие]
  Если для всякой интерпретации $\mathcal{I}$, такое что $[\![ \varphi ]\!]_{\mathcal{I}} = 1$, выполнено $[\![ \psi ]\!]_{\mathcal{I}} = 1$, то пишем $\varphi \models \varphi$.
\end{definition}

\begin{definition}
  $\mathcal{T}$ --- теория, если это множество формул.
\end{definition}

\begin{definition}
  \leavevmode
  \begin{enumerate}
  \item Теория $\mathcal{T}$ выполнима (SAT), если $\exists \ \mathcal{I}$ --- интерпретация, $\forall \ \varphi \in \mathcal{T}, [\![ \varphi ]\!]_{\mathcal{I}} = 1$.
  \item Теория $\mathcal{T}$ невыполнима (UNSAT), если $\forall \ \mathcal{I}$ --- интерпретация, $\exists \ \varphi \in \mathcal{T}, [\![ \varphi ]\!]_{\mathcal{I}} = 0$.
  \end{enumerate}
\end{definition}

\begin{definition}
  Пусть $\mathcal{T}$ --- теория, $\varphi$ --- формула.
  Мы пишем $\mathcal{T} \models \varphi$, если для любой интерпретации $\mathcal{I}$, при которой $\mathcal{T}$ выполнима, $[\![ \psi ]\!]_{\mathcal{I}} = 1$
\end{definition}

\begin{definition}[Модель теории]
  Пусть $\mathcal{T}$ --- теория, $\mathcal{I}$ --- интерпретация.

т
  $\mathcal{I}$ называется модель теории $\mathcal{T}$, если $\forall \varphi \in \mathcal{I}, [\![ \psi ]\!]_{\mathcal{I}} = 1$
\end{definition}

\begin{notation}
  Пусть $\mathcal{I}: \mathcal{P} \to \mathbb{B}_0$ --- интерпретация

  Тогда
  $Th(\mathcal{I}) = \{\varphi : [\![ \varphi ]\!]_{\mathcal{I}} = 1\}$
\end{notation}

\begin{corollary}
  \leavevmode
  \begin{enumerate}
  \item Теория $\mathcal{T}$ выполнима (SAT), если $\exists \ \mathcal{M}$ --- модель теории $\mathcal{T}$.
  \item Теория $\mathcal{T}$ невыполнима (UNSAT), если $\nexists$ модели.
  \end{enumerate}
\end{corollary}
\begin{remark}[Упрощение обозначения логического следствия]
  \leavevmode
  \begin{enumerate}
  \item $\{p, q\} \models r \iff p, q \models r$
  \item $T \cup \varphi \models \psi \iff T , \varphi \models \psi$
  \item $\varnothing \models \varphi \iff \ \models \varphi$
  \end{enumerate}
  Здесь $\iff$ --- равносильность утверждений
\end{remark}

\begin{definition}
  Пусть $A$ и $T$ --- теории.
  $A$ --- аксиома теории $T$,
  если $\forall \varphi \in A, T \models \varphi \ $  и  $ \ \forall \varphi \in T, A \models \varphi$
\end{definition}
\begin{definition}
 $T$  --- конечно аксиоматизируема, если $\exists A : A$ --- конечная аксиома теории $T$
\end{definition}

\begin{theorem}[О логическом следствии]
  Пусть $\Gamma$ --- множество формул, $\varphi$ и $\psi$ --- формулы. Тогда
  \[ \Gamma, \varphi \models \psi \hspace{1cm} \text{тогда и только тогда, когда} \hspace{1cm} \Gamma \models \varphi \to \psi\]
\end{theorem}
\begin{proof}
  \textbf{Необходимость}:

  Пусть $\Gamma, \varphi \models \psi$ и $\mathcal{I}$ --- модель $\Gamma$

  \underline{Случай 1}: $[\![ \varphi ]\!]_{\mathcal{I}} = 0$.
  Очевидно, т.к. из лжи следует всё что угодно

  \underline{Случай 2}: $[\![ \varphi ]\!]_{\mathcal{I}} = 1$.

  $[\![ \varphi ]\!]_{\mathcal{I}} = 1 \Rightarrow [\![ \psi ]\!]_{\mathcal{I}} = 1 \Rightarrow [\![ \varphi \to \psi ]\!]_{\mathcal{I}} = 1$

  \textbf{Достаточность}:

  Пусть $\Gamma \models \varphi \to \psi$ и $\mathcal{I}$ --- модель $\Gamma$

  Знаем, что $[\![ \varphi \to \psi ]\!]_{\mathcal{I}} = 1$

  Если выполнено $[\![ \varphi ]\!]_{\mathcal{I}} = 1$, тогда выполнено $[\![ \psi ]\!]_{\mathcal{I}} = 1$, что удовлетворяет $\Gamma, \varphi \models \psi$ по определению

\end{proof}

\begin{definition}
  $\varphi \sim \psi \iff \varphi \models \psi \ $ и $\ \psi \models \varphi$
\end{definition}
\begin{example}
  \leavevmode
  \begin{itemize}
  \item $\varphi \to \psi \sim \lnot \psi \to \lnot \varphi$
  \item $\lnot \lnot \varphi \sim \varphi$
  \item $\lnot (\varphi \land \psi) \sim \lnot \varphi \lor \lnot \psi$
  \end{itemize}
\end{example}

Исчисление --- это некая система над символами, дающая нам возможность дедуцировать слова из уже существующих слов, используя так называемые правила вывода.

\begin{example}
  Тут очень подошёл бы пример с SAT солвером
\end{example}

\begin{definition}
  Исчисление $\mathcal{C}$ задаётся:
  \begin{enumerate}
  \item Языком
  \item Аксиомами
  \item Правилами вывода
  \end{enumerate}
  Теперь по-подробнее:

  Пусть $L$ --- язык исчисления $\mathcal{C}$ над алфавитом $\Sigma$
  \begin{enumerate}
  \item Множество аксиом $Ax$--- это подмножество $L$
  \item
    Правило вывода $\mathcal{R}$ --- это $(n + 1)$-арное отношение, где $n \in \mathbb{N}_+$

    Если $(\alpha_1, \alpha_2, ..., \alpha_n, \alpha) \in \mathcal{R}$, то говорят $\alpha$ получился из $\alpha_1, \alpha_2, ..., \alpha_n$ применением правилом вывода $\mathcal{R}$ или $\alpha$ --- результат применения $\mathcal{R}$ из $\alpha_1, \alpha_2, ..., \alpha_n$

  \end{enumerate}
\end{definition}

Мы будем дальше говорить, что гипотеза $\Gamma$ --- это подмножество $L$

\begin{definition}
  Вывод из гипотезы $\Gamma$ является последовательность слов $\alpha_1, \alpha_2, ..., \alpha_n$ c запятой-разделителем в которой каждое слово $\alpha_i$...
  \begin{enumerate}
  \item Либо аксиома
  \item Либо гипотеза
  \item Либо получается по некоторому правилу выводу из прошлых выведенных слов $\alpha_{i_1}, \alpha_{i_2}, ..., \alpha_{i_k}$, где $i_1, ..., i_k $ меньше $i$
  \end{enumerate}

\end{definition}

\begin{notation}
  $\Gamma \vdash \varphi$ --- значит $\varphi$ выводится из гипотез, принадлежащих $\Gamma$

  Ещё $\varnothing \vdash \varphi \iff \vdash \varphi$
\end{notation}

\begin{definition}
  Исчисления высказываний Гильбертского типа --- это исчисление, которое строится следующим образом:
  \begin{enumerate}
  \item Язык --- это язык высказываний
  \item Аксиомы: %% (Дальше пояснения в скобках --- не просто мнемоника, а соответствие Чёрча-Ховарда (TODO: Проверить на достоверность))

    Пусть $\alpha, \beta, \gamma$ --- well formed formulas

    \begin{enumerate}
    \item $\alpha \rightarrow \beta \rightarrow \alpha$ (Как комбинатор K из SKI)
    \item $(\alpha \rightarrow \beta \rightarrow \gamma) \rightarrow (\alpha \rightarrow \beta) \rightarrow (\alpha \rightarrow \gamma)$ (Как комбинатор S из SKI)
    \item $\alpha \land \beta \rightarrow \alpha$ (Как функция fst из OCaml)
    \item $\alpha \land \beta \rightarrow \beta$ (Как функция snd из OCaml)
    \item $\alpha \rightarrow \alpha \lor \beta$ (Left)
    \item $\beta \rightarrow \alpha \lor \beta$ (Right)
    \item $\alpha \rightarrow \beta \rightarrow \alpha \land \beta$ (Создание пары)
    \item $(\alpha \rightarrow \gamma) \rightarrow (\beta \rightarrow \gamma) \rightarrow (\alpha \lor \beta \rightarrow \gamma)$ (Pattern matching)
    \item $\alpha \rightarrow \top$ ( fun x $\rightarrow$ () в OCaml)
    \item $\bot \rightarrow \alpha$ ( $\bot$ --- аналог type void = | )
    \item $(\alpha \rightarrow \bot) \rightarrow \lnot \alpha$
    \item $\lnot \alpha \rightarrow (\alpha \rightarrow \bot)$
    \item $(\alpha \rightarrow \beta) \rightarrow (\alpha \rightarrow \lnot \beta) \rightarrow \lnot \alpha$
    \item $\lnot \lnot \alpha \rightarrow \alpha$ (Снятие двойного отрицания)
    \end{enumerate}
  \item Правило вывода только одно --- modus ponens, а именно
  \begin{prooftree}
    \AxiomC{$\alpha, \ \alpha \rightarrow \beta$}
    \UnaryInfC{$\beta$}
  \end{prooftree}
  \end{enumerate}
\end{definition}

\end{example}

\begin{theorem}[Обобщённая теорема дедукции]
  $\Gamma$ --- множество формул

  $\varphi$, $\psi$ --- формулы. Тогда

  \[ \Gamma, \varphi \vdash \psi \hspace{1cm} \text{тогда и только тогда, когда} \hspace{1cm} \Gamma \vdash \varphi \to \psi\]
\end{theorem}
\begin{proof}
  \textbf{Достаточность}:

  Пусть $\Gamma, \vdash \varphi \rightarrow \psi$, тогда существует формальный вывод слова $\varphi \rightarrow \psi$:
  \[........, \varphi \rightarrow \psi\] Рассматриваем гипотезу $\Gamma$ вместе с $\varphi$, из гипотез выводится формальный вывод и к нему добавим $\varphi$ справа
  \[........, \varphi \rightarrow \psi, \varphi\]
  Применим modus ponens и получим $\psi$. Мы вывели $\psi$ из $\Gamma$ и $ \varphi$, т.е. $\Gamma, \varphi \vdash \psi$.

  \textbf{Необходимость}:

  Пусть $\Gamma, \varphi \vdash \psi$, тогда существует формальный вывод слова $\psi$:
  \[ \alpha_1 , \alpha_2, ..., \alpha_n\]
  , где $\alpha_n = \psi$

  Сделаем такую последовательность слов:
  \[ \varphi \rightarrow \alpha_1 , \varphi \rightarrow \alpha_2, ..., \varphi \rightarrow \alpha_n\]
  Хотим сделать этой последовательности формальный вывод, запустим процесс `вылечивания`.

  Рассматриваем все $\alpha_i$, для каждой --- следующие случаи:
  \begin{itemize}
  \item Если $\alpha_i = \varphi$, то $\varphi \rightarrow \varphi$ --- тавтология
  \item Если $\alpha_i$ --- аксиома или гипотеза, тогда заменяем $\varphi \rightarrow \alpha_i$  на $\alpha_i \rightarrow \varphi \rightarrow \alpha_i$ (это верно по аксиоме), применяем modus ponens и получаем $\varphi \rightarrow \alpha_i$
  \item Если $\alpha_i$ --- выводится

    В изначальном выводе было
    \[ ... ,\beta \to \alpha_i, ..., \beta, ..., \alpha_i\]

    Стало так
    \[ ... ,\varphi \to \beta \to \alpha_i, ..., \varphi \to \beta, ..., \varphi \to \alpha_i\]

    Добавляем аксиому типа S, применяем modus ponens два раза.

  \end{itemize}
\end{proof}
\begin{theorem}[Теорема корректности]
  Если $T \vdash \varphi$, то $T \models \varphi$
\end{theorem}
\begin{proof}
  Индукция по длине вывода $n$:

  \textit{База : $n = 1$} \hfill \\
  Это значит, что $\varphi$ --- аксиома или гипотеза. Осталось проверить, что все аксиомы --- семантические тавтологии...

  \textit{Индукционный переход : $<n \to n$} \hfill \\
  Мы предпологаем, что теорема выполняется для любого слова, длина формального вывода которого меньше $n$.
  Докажем, что верно и для $n$. Рассмотрим вывод $T \vdash \varphi$ длины $n$.
  Случай с аксиомой и гипотезой мы уже разобрали. Тогда рассмотрим, что $\varphi$ --- выводится, т.е. следующий вывод
  \[ ....., \alpha \to \varphi, ..., \alpha, ..., \varphi\]

  Рассмотрим этот вывод без $\varphi$. Применяя индукционный переход получим, что

  \[ [\![ \alpha \to \varphi ]\!] = [\![ \alpha ]\!] = 1 \]

  Тогда понятно, что $[\![ \varphi ]\!] = 1$

\end{proof}

\begin{definition}
  Теория $\mathcal{T}$ называется полной (complete), если
  \begin{enumerate}
  \item Теория выполнима
  \item $\forall \varphi$ такие что $\varphi \in \mathcal{T}$ или $\lnot \varphi \in \mathcal{T}$ выполняется $T \models \varphi$ или $T \models \lnot \phi$
  \end{enumerate}
\end{definition}

До конца главы нам нужно доказать теорему полноты (Если $T \models \varphi$, то $T \vdash \varphi$), но для этого нужно доказать две вспомогательных теоремы.

\begin{definition}
  Теория $\mathcal{T}$ называется непротиворечивой (consistent), если
  для любой формулы $\varphi \in \mathcal{T}$ нельзя вывести её и её отрицание.
\end{definition}

\begin{remark}
  Если теория $\mathcal{T}$ противоречивая, то можно вывести $\bot$
\end{remark}

\begin{definition}
  Теория $\mathcal{T}$ называется максимально непротиворечивой (max consistent), если
  для любой формулы $\varphi \in \mathcal{T}$ можно вывести либо её, либо её отрицание.
\end{definition}

\begin{remark}
  Она \textbf{максимально} непротиворечивая, т.к уже нельзя расширить теорию, чтобы она ещё оставалось выполняемой
\end{remark}

\begin{remark}
  Из максимальной непротиворечивости следует непротиворечивость.
\end{remark}

\begin{theorem}[Лемма Линденбаума]
  Всякая непротиворечивая теория содержится в некоторой максимально непротиворечивой теории
\end{theorem}
\begin{proof}
  Для счётного языка можно пронумеровать все формулы (смотри Gödel numbering). Пусть $\mathcal{T} = \{\alpha_1, \alpha_2, ...\}$
  Дальше определяем следующую SAT теорию:
  \[
  T_0 = \top
  \]
  \[
  T_{n+1} =\begin{cases}
    \{\alpha_n\} \cup T_n & ,\text{если она SAT}\\
    T_n & ,\text{иначе}
  \end{cases}
  \]

  Определяем \[T^* = \bigcup\limits_{n=0}^{\infty}T_n\]

  \begin{description}
  \item[$T^* \text{ непротиворечивая:}$] \hfill \\
    Предположим обратное, пусть $T^*$ --- противоречивая теория, тогда есть формальный вывод $\bot$. Это формальный вывод пренадлежит также конечному подмножеству $T^*$, назовём такую теорию $T^{\prime}$. Тогда $\exists n$, что $T^{\prime} \subset T_n$, но тогда $T^{\prime}$ непротиворечивая (?!!)
  \item[$T^* \text{ максимально непротиворечивая:}$] \hfill \\
    Предположим обратное, пусть $\exists \varphi : T^* \nvdash \varphi$ и $T^* \nvdash \lnot \varphi$. Докажем, что $T^* \cup \{\varphi\}$ --- непротиворечивая. Опять от противного, $T^* \cup \{\varphi\}$ --- противоречивая, тогда \[T^*, \varphi \vdash \bot\]
    Применяя теорему о дедукции, получаем $T^*\vdash \varphi \to \bot$, но если воспользоваться аксиомой
    $(\varphi \to \bot) \to \lnot \varphi$, то тогда $T^*\vdash \lnot \varphi$ (?!!)
  \end{description}
\end{proof}
\begin{theorem}[о $\exists$ модели]
  Если теория непротиворечивая, то для неё существует модель
\end{theorem}
\begin{proof}
  По предыдущей теореме существует максимально непротиворечивая теория $T^* \subset T$. Любая модель $T^*$ есть $T$, поэтому для доказательства теоремы достаточно найти модель для $T^*$.

  $T^*$ --- максимально непротиворечивая, тогда для любой пропозициональной переменной $p$ либо $T^* \vdash p$, либо $T^* \vdash \lnot p$.

  Определим
  \begin{align}
    \mathcal{I}(p) = \begin{cases}
      1, & T^* \vdash p\\
      0, & T^* \vdash \lnot p
    \end{cases}
  \end{align}


  Докажем по структурной индукции, что $\mathcal{I}$ --- модель $T^*$

  \begin{description}
  \item[$\text{База}:$]
    $p$ --- пропозиционная переменная,
    $[\![ p ]\!] = 1 \iff T^* \vdash p$
  \item[$\text{Переход}:$]
    Пусть для $\alpha, \beta$ выполняется доказываемое утверждение
    Нужно доказать утверждение для $\alpha \land \beta, \alpha \lor \beta, \alpha \to \beta, \lnot \alpha$.
    \begin{enumerate}
    \item Рассмотрим $\lnot \alpha$. $[\![ \alpha ]\!] = 0 \iff [\![ \lnot \alpha ]\!] = 1$. По индукционному предположение ясно, что $T^* \vdash \alpha \iff [\![ \alpha ]\!] = 0$. Пользуясь непротиворечивостью и теоремой о полноте получаем, что $T^* \nvdash \alpha \iff T^* \vdash \lnot \alpha$
    \end{enumerate}
    Дальше аналогичные рассуждение для других трёх пунктов

  \end{description}
\end{proof}
\begin{theorem}[Теорема о полноте]
Если $T \models \varphi$, то $T \vdash \varphi$
\end{theorem}
\begin{proof}
  Пусть $T \models \varphi$, тогда $T \cup \{\lnot \varphi\}$ --- невыполнимая теория, соответсвенно,
  \[T \vdash \lnot \varphi \to \bot\]

  Тогда существует формальный вывод $\lnot \varphi \to \bot$,
  добавим аксиому $(\lnot \varphi \to \bot) \to \lnot \lnot \varphi$
  и применим modus ponens, выведем $\lnot \lnot \varphi$. Применим аксиому снятия двойного отрицания, получим, что требовалось доказать.

\end{proof}

\section{Генценовское секвенциальное исчисление (Sequent calculus)}
\begin{definition}
  Секвенция --- это слова вида $\Gamma \vdash \Delta$, где $\Gamma, \Delta$ --- конечные последовательности формул.

  $\Gamma$ называется \textbf{антецедентом}, а $Delta$ --- \textbf{сукцедентом}.
\end{definition}

\begin{definition}
  Генценовское секвенциальное исчисление определяется следующим образом
  \begin{enumerate}
  \item Язык --- это язык секвенций
  \item Аксиомы:
    Пусть $\varphi$ --- формула
    \begin{itemize}
    \item $\Gamma, \varphi \vdash \Delta, \varphi$
    \item $\Gamma, \bot \vdash \Delta$
    \item $\Gamma \vdash \top, \Delta$
    \end{itemize}
  \end{enumerate}
\item Правила вывода:
  \begin{enumerate}
  \item $LK_0$ правила:
    \begin{itemize}
      \begin{prooftree}
        \AxiomC{$\Gamma, \alpha \vdash \Delta \quad \Gamma, \alpha \vdash \Delta$}
        \UnaryInfC{$\Gamma, \alpha \lor \beta \vdash \Delta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma, \alpha, \beta \vdash \Delta$}
        \UnaryInfC{$\Gamma, \alpha \land \beta \vdash \Delta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma, \beta \vdash \Delta \quad \Gamma \vdash \alpha, \Delta$}
        \UnaryInfC{$\Gamma, \alpha \to \beta \vdash \Delta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma \vdash \alpha, \Delta $}
        \UnaryInfC{$\Gamma, \lnot \alpha \vdash \Delta$}
      \end{prooftree}
    \end{itemize}
    \item $NK_0$ правила:
    \begin{itemize}
      \begin{prooftree}
        \AxiomC{$\Gamma \vdash \alpha, \beta, \Delta$}
        \UnaryInfC{$\Gamma \vdash \alpha \lor \beta, \Delta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma \vdash \alpha \Delta \quad \Gamma \vdash \beta \Delta$}
        \UnaryInfC{$\Gamma \vdash \alpha \land \beta \Delta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma, \alpha \vdash \Delta \beta$}
        \UnaryInfC{$\Gamma \vdash, \alpha \to \beta$}
      \end{prooftree}
      \begin{prooftree}
        \AxiomC{$\Gamma , \alpha \vdash  \Delta $}
        \UnaryInfC{$\Gamma \vdash \lnot \alpha, \Delta$}
      \end{prooftree}
    \end{itemize}
  \end{enumerate}
\end{definition}
